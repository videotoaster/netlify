JS hover styling 
     function rollover(obj) {
       switch (obj.className) {
         case "alert":
           obj.style.background = "brown";
           obj.style.color = "white";
           break;
         case "success":
           obj.style.background = "green";
           obj.style.color = "white";
           break;
         case "error":
           obj.style.background = "maroon";
           obj.style.color = "white";
           break;
         case "info":
           obj.style.background = "navy";
           obj.style.color = "white";
           break;
         default:
           obj.style.background = "#444";
           obj.style.color = "white";
           break;
       }
     }
     function rolloff(obj) {
